```js
/**
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
// 一句话思路：用一个栈记录非零元素。将nums原地当作栈，将不为0的元素入栈，最后补0到length就行了
var moveZeroes = function (nums) {
  let stackSize = 0
  for (const n of nums) {
    if (n) {
      nums[stackSize++] = n
    }
  }
  for (let i = stackSize; i < nums.length; i++) {
    nums[i] = 0
  }
}
```
