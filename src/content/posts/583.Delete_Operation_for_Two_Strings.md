class Solution:
    def minDistance(self, word1: str, word2: str) -> int:
        n = len(word1)
        m = len(word2)
        @cache
        def dfs(i, j):
            if i < 0 or j < 0:
                return 0
            if word1[i] == word2[j]:
                return dfs(i-1, j-1) + 1
            return max(dfs(i, j-1), dfs(i-1, j))
        a = dfs(n-1, m-1)
        return n-a+(m-a)
