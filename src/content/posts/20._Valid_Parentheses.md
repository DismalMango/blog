注意三个cases

```python
class Solution:
    def isValid(self, s: str) -> bool:
        stack = []
        n = len(s)
        if n % 2 != 0:
            return False
        for i in s:
            if i in ['(', '{', '[']:
                stack.append(i)
            else:
                if stack == []:
                    return False
                p = stack.pop()
                if p == '(' and i != ')':
                    return False
                elif p == '{' and i != '}':
                    return False
                elif p == '[' and i != ']':
                    return False
        return True if stack == [] else False
```
